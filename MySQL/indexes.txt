-- USERS
CREATE UNIQUE INDEX idx_users_email ON users(email);
CREATE INDEX idx_users_user_type ON users(user_type);

-- STUDENTS
CREATE UNIQUE INDEX idx_students_student_number ON students(student_number);
CREATE INDEX idx_students_surname_name ON students(surname, name);

-- PROFESSORS
CREATE INDEX idx_professors_department ON professors(department);
CREATE INDEX idx_professors_surname_name ON professors(surname, name);
CREATE INDEX idx_professors_email ON professors(email);

-- THESES
CREATE INDEX idx_theses_supervisor_status_start ON theses(supervisor_id, status, start_date);
CREATE INDEX idx_theses_student ON theses(student_id);
CREATE INDEX idx_theses_status_start ON theses(status, start_date);
CREATE INDEX idx_theses_start_date ON theses(start_date);
CREATE INDEX idx_theses_end_date ON theses(end_date);
CREATE FULLTEXT INDEX idx_theses_title_description ON theses(title, description);

-- COMMITTEES
CREATE INDEX idx_committees_thesis ON committees(thesis_id);
CREATE INDEX idx_committees_supervisor ON committees(supervisor_id);
CREATE INDEX idx_committees_member1 ON committees(member1_id);
CREATE INDEX idx_committees_member2 ON committees(member2_id);

-- COMMITTEEINVITATIONS
CREATE INDEX idx_cinv_prof_status_time ON committeeinvitations(invited_professor_id, status, sent_at);
CREATE INDEX idx_cinv_thesis ON committeeinvitations(thesis_id);
CREATE INDEX idx_cinv_sent_at ON committeeinvitations(sent_at);

-- PROFESSORS_NOTIFICATIONS
CREATE INDEX idx_pn_prof_status_time ON professors_notifications(professor_id, status, sent_at);
CREATE INDEX idx_pn_thesis ON professors_notifications(thesis_id);
CREATE INDEX idx_pn_student ON professors_notifications(student_id);
CREATE INDEX idx_pn_sent_at ON professors_notifications(sent_at);

-- EXAMINATIONS
CREATE INDEX idx_exams_date_time ON examinations(exam_date, exam_time);
CREATE INDEX idx_exams_mode ON examinations(exam_mode);

-- EXAM_RESULTS
CREATE UNIQUE INDEX idx_er_thesis_prof ON exam_results(thesis_id, professor_id);
CREATE INDEX idx_er_thesis ON exam_results(thesis_id);
CREATE INDEX idx_er_professor ON exam_results(professor_id);
CREATE INDEX idx_er_exam_date ON exam_results(exam_date);

-- PROFESSOR_NOTES
CREATE INDEX idx_pnotes_thesis_time ON professor_notes(thesis_id, created_at);
CREATE INDEX idx_pnotes_prof ON professor_notes(professor_id);
CREATE INDEX idx_pnotes_prof_thesis_time ON professor_notes(professor_id, thesis_id, created_at);

-- ATTACHMENTS
CREATE INDEX idx_att_thesis_time ON attachments(thesis_id, uploaded_at);
CREATE INDEX idx_att_student_time ON attachments(student_id, uploaded_at);

-- ANNOUNCEMENTS
CREATE INDEX idx_ann_date_time ON announcements(date, time);
CREATE FULLTEXT INDEX idx_ann_fulltext ON announcements(title, announcement_text);

-- GRAMMATEIA
CREATE INDEX idx_gram_full_name ON grammateia(full_name);
